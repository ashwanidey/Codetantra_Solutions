# Python program to find the sum of integers between 0 and n where n is provided by user
num=int(input("num: "))
sum=0
if (num>0):
	while(num>=0):
		sum=sum+num
		num-=1
else:
	while num<0:
		sum=sum+num
		num+=1
print("sum:",sum)
